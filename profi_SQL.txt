1) 
кол-во действий за определённый месяц (скажем февраль 2020);
---------------
SELECT COUNT(*)
FROM actions
WHERE MONTH(dt) = 2 AND YEAR(dt) = 2020;
---------------

кол-во сотрудников, которые выполняли хоть какие-нибудь действия за указанный период;
---------------
SELECT COUNT(*)
FROM actions
WHERE action IS NOT NULL; 
---------------


2) Составить запрос, который выводит логины сотрудников, которые не выполняли никаких действий в феврале 2020.
---------------
SELECT e.login
FROM employees AS e
INNER JOIN actions AS a
WHERE a.action IS NULL;
---------------

3) Составить запрос, который выводит 5 сотрудников с наибольшим количеством действий за февраль 2020, а также кол-во их действий, отсортированных в порядке убывания.
---------------
SELECT e.id, e.login
FROM employees AS e
WHERE e.id IN (SELECT employee_id, COUNT(action)
               FROM actions
               WHERE MONTH(dt) = 2 AND YEAR(dt) = 2020
               GROUP BY employee_id
               ORDER BY COUNT(action) DESC)
LIMIT 5;
---------------
